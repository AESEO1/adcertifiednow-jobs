# Chrome Web MCP Integration Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add 10 Chrome Web MCP (Gemini Nano) AI features across the jobsite with zero API cost, as pure progressive enhancement for Chrome 127+ users.

**Architecture:** A shared `lib/ai.ts` utility wraps `window.ai` and is imported by focused single-purpose components in `components/ai/`. All components return `null` when AI is unavailable — non-Chrome users see no difference. Job detail page gets summary, CV matcher, skills, salary, remote, and seniority enrichment. Search gets natural language parsing and query expansion. Post-job form gets title normaliser and description assistant.

**Tech Stack:** Next.js 14, React 18, TypeScript 5, Tailwind CSS, shadcn/ui (Badge, Button, Card, Textarea, Separator), Chrome Prompt API (`window.ai`)

```ts
export async function isAIAvailable(): Promise<boolean> {
  if (typeof window === 'undefined' || !window.ai) return false
  try {
    const availability = await window.ai.languageModel.availability()
    return availability === 'available'
  } catch {
    return false
  }
}

export async function promptAI(prompt: string): Promise<string> {
  const session = await window.ai!.languageModel.create()
  try {
    return await session.prompt(prompt)
  } finally {
    session.destroy()
  }
}

export async function streamAI(prompt: string): Promise<ReadableStream> {
  const session = await window.ai!.languageModel.create()
  const stream = session.promptStreaming(prompt)
  // Wrap so we can destroy session when stream ends
  const reader = stream.getReader()
  return new ReadableStream({
    async pull(controller) {
      const { done, value } = await reader.read()
      if (done) {
        controller.close()
        session.destroy()
      } else {
        controller.enqueue(value)
      }
    },
    cancel() {
      reader.cancel()
      session.destroy()
    },
  })
}
```

```tsx
'use client'

import { useEffect, useState, useRef } from 'react'
import { Card, CardContent, CardHeader } from '@/components/ui/card'
import { isAIAvailable, streamAI } from '@/lib/ai'

interface JobSummaryProps {
  description: string
}

export default function JobSummary({ description }: JobSummaryProps) {
  const [bullets, setBullets] = useState<string[]>([])
  const [loading, setLoading] = useState(false)
  const [show, setShow] = useState(false)
  const cancelRef = useRef(false)

  useEffect(() => {
    cancelRef.current = false
    let reader: ReadableStreamDefaultReader | null = null

    async function run() {
      const available = await isAIAvailable()
      if (!available || cancelRef.current) return
      setShow(true)
      setLoading(true)

      try {
        const stream = await streamAI(
          `Summarise this job description in exactly 3 bullet points, each under 15 words. Return as a plain numbered list like "1. ...", no intro text: ${description.substring(0, 3000)}`
        )
        reader = stream.getReader()
        let text = ''
        while (true) {
          const { done, value } = await reader.read()
          if (done || cancelRef.current) break
          text += value
          // Parse lines as they stream in
          const lines = text
            .split('\n')
            .map(l => l.replace(/^\d+\.\s*/, '').trim())
            .filter(Boolean)
          setBullets(lines.slice(0, 3))
        }
      } catch {
        // silent failure
      } finally {
        setLoading(false)
      }
    }

    run()
    return () => {
      cancelRef.current = true
      reader?.cancel()
    }
  }, [description])

  if (!show) return null

  return (
    <Card>
      <CardHeader>
        <h2 style={{ display: 'flex', alignItems: 'center', gap: '8px', fontSize: '18px', fontWeight: 'bold', color: '#111827', margin: 0 }}>
          <span>✨</span>
          <span>AI Summary</span>
        </h2>
      </CardHeader>
      <CardContent>
        {loading && bullets.length === 0 ? (
          <div style={{ display: 'flex', flexDirection: 'column', gap: '8px' }}>
            {[1, 2, 3].map(i => (
              <div key={i} style={{ height: '16px', backgroundColor: '#f3f4f6', borderRadius: '4px', animation: 'pulse 2s infinite' }} />
            ))}
          </div>
        ) : (
          <ul style={{ display: 'flex', flexDirection: 'column', gap: '8px', margin: 0, paddingLeft: '20px', color: '#374151' }}>
            {bullets.map((b, i) => (
              <li key={i}>{b}</li>
            ))}
          </ul>
        )}
      </CardContent>
    </Card>
  )
}
```

```tsx
import { Badge } from '@/components/ui/badge'

  description: string

  useEffect(() => {
    let cancelled = false
    async function run() {
      const available = await isAIAvailable()

  if (skills.length === 0) return null

  return (
    <>
      <Separator style={{ margin: '4px 0' }} />
      {skills.map(skill => (
        <Badge key={skill} variant="secondary" style={{ fontSize: '13px' }}>
          {skill}

    </>
  )
}
```

```tsx
import { Badge } from '@/components/ui/badge'

const VALID = ['Junior', 'Mid', 'Senior', 'Lead']

const COLOURS: Record<string, string> = {
  Junior: '#dcfce7',
  Mid: '#dbeafe',
  Senior: '#fef9c3',
  Lead: '#fce7f3',

  description: string

export default function SeniorityBadge({ title, description }: SeniorityBadgeProps) {

        </Badge>
```

```tsx
export function CVMatcherButton() {

export function CVMatcherCard({ jobDescription, jobTitle }: CVMatcherProps) {
  const [show, setShow] = useState(false)
  const [cv, setCv] = useState('')
  const [result, setResult] = useState('')
  const [loading, setLoading] = useState(false)
  const cancelRef = useRef(false)

    isAIAvailable().then(setShow)

  if (!show) return null

    cancelRef.current = false
    setLoading(true)
    setResult('')

      const stream = await streamAI(

${jobDescription.substring(0, 3000)}`

        if (done || cancelRef.current) break
        text += value
        setResult(text)
      }
    } catch {
      setResult('Unable to analyse CV — please try again.')
    } finally {
      setLoading(false)
    }

  return (
```

```tsx
'use client'

import { useState, useEffect, useRef } from 'react'
import { useRouter } from 'next/navigation'
import { Input } from '@/components/ui/input'
import { Button } from '@/components/ui/button'
import { isAIAvailable, promptAI } from '@/lib/ai'

interface AISearchBarProps {
  defaultValue?: string
  className?: string
  inputClassName?: string
  buttonClassName?: string
}

interface ParsedQuery {
  keywords: string
  location: string | null
  category: string | null
  contractType: string | null
  minSalary: number | null
}

export default function AISearchBar({
  defaultValue = '',
  className,
  inputClassName,
  buttonClassName,
}: AISearchBarProps) {
  const router = useRouter()
  const [aiMode, setAiMode] = useState(false)
  const [aiAvailable, setAiAvailable] = useState(false)
  const [query, setQuery] = useState(defaultValue)
  const [processing, setProcessing] = useState(false)

  useEffect(() => {
    isAIAvailable().then(setAiAvailable)
  }, [])

  async function expandQuery(keyword: string): Promise<string> {
    try {
      const raw = await promptAI(
        `List up to 4 common job title synonyms for '${keyword}'. Return only a JSON array of short strings.`
      )
      const parsed = JSON.parse(raw.trim().replace(/```json|```/g, ''))
      if (Array.isArray(parsed)) {
        return [keyword, ...parsed].join(' ')
      }
    } catch { /* silent */ }
    return keyword
  }

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault()
    if (!query.trim()) return
    setProcessing(true)

    try {
      if (aiMode && aiAvailable) {
        const raw = await promptAI(
          `Parse this job search query into JSON with keys: keywords (string), location (string or null), category (string or null), contractType (string or null), minSalary (number or null). Query: ${query}`
        )
        const parsed: ParsedQuery = JSON.parse(raw.trim().replace(/```json|```/g, ''))
        const expanded = await expandQuery(parsed.keywords)
        const params = new URLSearchParams()
        if (parsed.location) params.set('city', parsed.location)
        if (parsed.contractType) params.set('type', parsed.contractType)
        router.push(`/search/q/${encodeURIComponent(expanded)}?${params.toString()}`)
      } else if (aiAvailable) {
        const expanded = await expandQuery(query.trim())
        router.push(`/search/q/${encodeURIComponent(expanded)}`)
      } else {
        router.push(`/search/q/${encodeURIComponent(query.trim())}`)
      }
    } catch {
      router.push(`/search/q/${encodeURIComponent(query.trim())}`)
    } finally {
      setProcessing(false)
    }
  }

  return (
    <div className={className}>
      <form onSubmit={handleSubmit} style={{ display: 'flex', gap: '16px' }}>
        <Input
          type="text"
          value={query}
          onChange={e => setQuery(e.target.value)}
          placeholder={
            aiMode
              ? 'e.g. part-time nurse jobs Manchester over 30k...'
              : 'Search job titles...'
          }
          className={inputClassName ?? 'flex-1 px-4 py-3 rounded-lg text-gray-900'}
          required
        />
        <Button
          type="submit"
          disabled={processing}
          className={buttonClassName ?? 'bg-orange-500 hover:bg-orange-600 px-8 py-3 rounded-lg font-semibold text-white'}
        >
          {processing ? 'Searching...' : 'Search Jobs'}
        </Button>
      </form>
      {aiAvailable && (
        <div style={{ marginTop: '8px', textAlign: 'center' }}>
          <button
            type="button"
            onClick={() => setAiMode(m => !m)}
            style={{
              background: 'none',
              border: 'none',
              cursor: 'pointer',
              fontSize: '13px',
              color: aiMode ? '#f97316' : 'rgba(255,255,255,0.8)',
              textDecoration: 'underline',
            }}
          >
            {aiMode ? '✨ AI search on — switch to standard' : '✨ Try AI search'}
          </button>
        </div>
      )}
    </div>
  )
}
```

## Task 10: JobDescriptionAssistant component (Feature 10)

- [ ] **Step 1: Create `components/ai/JobDescriptionAssistant.tsx`**

```tsx
'use client'

import { useEffect, useState, useRef } from 'react'
import { isAIAvailable, promptAI } from '@/lib/ai'

const ALL_ELEMENTS = ['salary', 'location', 'requirements', 'benefits', 'contract type', 'application instructions']

interface JobDescriptionAssistantProps {
  description: string
}

export default function JobDescriptionAssistant({ description }: JobDescriptionAssistantProps) {
  const [missing, setMissing] = useState<string[]>([])
  const [aiReady, setAiReady] = useState(false)
  const debounceRef = useRef<ReturnType<typeof setTimeout> | null>(null)

  useEffect(() => {
    isAIAvailable().then(setAiReady)
  }, [])

  useEffect(() => {
    if (!aiReady || !description.trim()) return
    if (debounceRef.current) clearTimeout(debounceRef.current)
    debounceRef.current = setTimeout(async () => {
      try {
        const raw = await promptAI(
          `Review this job description and list which of these elements are missing: salary, location, requirements, benefits, contract type, application instructions. Return as a JSON array of missing item names only, e.g. ["salary","benefits"]. Description: ${description.substring(0, 3000)}`
        )
        const parsed = JSON.parse(raw.trim().replace(/```json|```/g, ''))
        if (Array.isArray(parsed)) {
          setMissing(parsed.filter((x: unknown) => typeof x === 'string' && ALL_ELEMENTS.includes(x)))
        }
      } catch { /* silent */ }
    }, 2000)

    return () => {
      if (debounceRef.current) clearTimeout(debounceRef.current)
    }
  }, [description, aiReady])

  if (!aiReady || missing.length === 0) return null

  return (
    <div style={{ marginTop: '8px' }}>
      <p style={{ fontSize: '12px', color: '#6b7280', marginBottom: '6px' }}>✨ Consider adding:</p>
      <div style={{ display: 'flex', flexWrap: 'wrap', gap: '6px' }}>
        {missing.map(item => (
          <span
            key={item}
            style={{
              fontSize: '12px',
              padding: '2px 8px',
              backgroundColor: '#fef9c3',
              border: '1px solid #fde68a',
              borderRadius: '12px',
              color: '#92400e',
            }}
          >
            {item}
          </span>
        ))}
      </div>
    </div>
  )
}
```
